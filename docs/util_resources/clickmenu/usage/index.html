Usage

IMenu Interface‚Äã

name string - Menu entry name to show‚Äã

icon? string - Icon to show (should use from fontawesome)‚Äã

eventName? string - Event name to trigger‚Äã

args? any - Event argumentums‚Äã

env? "server" | "client" - Which environment trigger the event?‚Äã

description? string - Menu entry description‚Äã

Clientside‚Äã

Open‚Äã

atCursor? boolean Open menu at cursor coordinates‚Äã

exports["avp_click_menu"]:open(atCursor)

Close‚Äã

exports["avp_click_menu"]:close()

SetData‚Äã

---@param menuHeader string
---@param menuData IMenu[]
exports["avp_click_menu"]:setData(menuHeader, menuData)

Serverside‚Äã

Open‚Äã

exports["avp_click_menu"]:open(source, atCursor)

Close‚Äã

exports["avp_click_menu"]:close(source)

SetData‚Äã

---@param source number | string
---@param menuHeader string
---@param menuData IMenu[]
exports["avp_click_menu"]:setData(source, menuHeader, menuData)

Examples‚Äã

Serverside‚Äã

RegisterNetEvent("Ping", function(response)
    local source = source

    print(source .. " " .. response)
end)

RegisterCommand("sv_test", function(source)
    exports["avp_click_menu"]:setData(
        source,
        "Vehicle (Server)",
        {
            {
                icon = "fa-solid fa-car",
                name = "Open door",
                eventName = "Ping",
                env = "server",
                args = "Pong",
                description = "Description (server)"
            }
        }
    )
    exports["avp_click_menu"]:open(source)
end)

Clientside‚Äã

AddEventHandler("Ping", function(response)
    print(response) -- Pong
end)

RegisterCommand("cl_test", function()
    exports["avp_click_menu"]:setData(
        "Vehicle (Client)",
        {
            {
                icon = "fa-solid fa-car",
                name = "Open door",
                eventName = "Ping",
                env = "client",
                args = "Pong",
                description = "Description (client)"
            }
        }
    )
    exports["avp_click_menu"]:open(true)
end)
